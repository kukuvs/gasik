# ER
```
User {
	id integer pk increments unique
	name char(100)
	Phone integer(20)
	mail char(250)
	age integer(2)
	raiting integer(5)
	skills integer(10)
	notifications bool
	password integer(30)
}

Skils {
	id integer pk increments unique >* Skil_user.id
	title char(30)
}

Skil_user {
	id integer pk increments unique
	id_user integer > User.id
}

Events {
	id integer pk increments unique >* Events_user.id
	title integer(100)
	description char(2000)
	start date
	end date
}

Events_user {
	id integer pk increments unique
	id_user integer > User.id
}

Projects {
	id integer pk increments unique >* Projects_user.id
	title char(30)
	description char(1000)
	date_proj date
	Url char(255)
	main_user integer > User.id
}

Projects_user {
	id_user integer > User.id
	id integer pk increments unique
}

Corp {
	name char
	description char
	mail char
	password integer
}
```
